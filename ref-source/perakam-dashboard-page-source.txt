<script>
//running clock
var today = moment('2026-06-24 15:08:10', 'YYYY-MM-DD HH:mm:ss').toDate();
var d = today.getDate();
var b = moment(today).format('MMM');
var y = today.getFullYear();
var h = today.getHours();
var m = today.getMinutes();
var s = today.getSeconds();

$(document).ready(function() {
	startTime();
});

function startTime() {
    hh = checkTime(h,'hour');
    mm = checkTime(m,'minute');
    ss = checkTime(s,'second');
    pp = AMPM(h);
	
	document.getElementById('runtheclock').innerHTML =
    checkTime(d,'day') + " " + b.toUpperCase() + " " + y + " " + hh + ":" + mm + ":" + ss + " " + pp; 
	
	s++;
	
	if (s==60) { s=0; m++; }
	if (m==60) { m=0; h++; }
	if (h==24) { h=0; }
	
    var t = setTimeout(startTime, 1000);
}
function checkTime(i,fmt) {
    if (i > 12 && fmt=='hour') { i -= 12; }
    if (i < 10) { i = "0" + i; }  // add zero in front of numbers < 10
    return i;
}
function AMPM(j) {
	if (j<12){
		return "AM";
	}else{
		return "PM";
	}
}
